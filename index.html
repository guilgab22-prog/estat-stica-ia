document.getElementById('btn').addEventListener('click', async () => { if(!csvTexto) return alert("Carregue o CSV primeiro."); // O CADEADO: Desliga o botão imediatamente após o primeiro toque const btn = document.getElementById('btn'); btn.disabled = true; btn.style.background = "#ccc"; btn.innerText = "A processar..."; status.innerText = "A pedir código à IA (Servidor Seguro)..."; resultado.style.display = 'none'; document.getElementById('grafico').style.display = 'none'; try { const resAPI = await fetch('/api/gemini', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ prompt: document.getElementById('prompt').value, csvText: csvTexto }) }); const dadosAPI = await resAPI.json(); if(dadosAPI.error) throw new Error(dadosAPI.error); status.innerText = "A renderizar matemática e gráficos..."; await webr.FS.writeFile('/d.csv', new TextEncoder().encode(csvTexto)); const sep = csvTexto.split('\n')[0].includes(';') ? ';' : ','; const cmdLer = `d <- read.csv${sep===';'?'2':''}('/d.csv')`; const script = `library(ggplot2)\n${cmdLer}\n${dadosAPI.codigo}`; const run = await webr.evalR(`png('/g.png', width=800, height=500, res=120); out <- capture.output({eval(parse(text='${script.replace(/'/g, "\\'").replace(/\n/g, "\\n")}'))}); dev.off(); out`); const vals = await run.toArray(); resultado.innerText = `Saída de Texto:\n${vals.join('\n')}`; resultado.style.display = 'block'; try { const img = await webr.FS.readFile('/g.png'); document.getElementById('grafico').src = URL.createObjectURL(new Blob([img], {type:'image/png'})); document.getElementById('grafico').style.display = 'block'; } catch(e) {} status.innerText = "Análise concluída!"; } catch(e) { status.innerText = "Falha na análise."; status.style.color = "red"; resultado.innerText = "Erro do Servidor: " + e.message; resultado.style.display = 'block'; } finally { // TIRA O CADEADO: Volta a ligar o botão no final, dê erro ou sucesso btn.disabled = false; btn.style.background = "#0056b3"; btn.innerText = "Executar Análise"; } });
